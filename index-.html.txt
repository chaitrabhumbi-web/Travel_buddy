<script>
  function showScreen(id) {
    document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
    document.getElementById(id).classList.add('active');
    window.scrollTo(0, 0);
  }

  function switchTab(screenId, btn) {
    showScreen(screenId);
    // Update all bottom navs
    document.querySelectorAll('.nav-tab').forEach(t => t.classList.remove('active'));
    document.querySelectorAll('.bottom-nav').forEach(nav => {
      const tabs = nav.querySelectorAll('.nav-tab');
      tabs.forEach(t => {
        if (t.getAttribute('onclick') && t.getAttribute('onclick').includes(screenId)) {
          t.classList.add('active');
        }
      });
    });
  }

  function toggleTag(btn) {
    btn.classList.toggle('selected');
  }

  function goOnboardStep2() {
    document.getElementById('onboard-s1').classList.remove('active');
    document.getElementById('onboard-s2').classList.add('active');
    document.getElementById('onboard-title').textContent = 'Where do you\nwant to go? 🗺️';
    document.getElementById('onboard-subtitle').textContent = 'Pick dream destinations & travel timeline';
    const dots = document.querySelectorAll('#onboard-step-dots .step-dot');
    dots[2].classList.add('done');
  }

  function likePost(btn) {
    if (btn.classList.contains('liked')) {
      btn.classList.remove('liked');
      btn.innerHTML = '<span class="icon">🤍</span>' + btn.innerHTML.replace(/❤️|🤍/, '').trim();
    } else {
      btn.classList.add('liked');
      btn.innerHTML = '<span class="icon">❤️</span>' + btn.innerHTML.replace(/❤️|🤍/, '').trim();
    }
  }

  function connectBuddy(btn) {
    if (btn.classList.contains('connected')) return;
    btn.classList.add('connected');
    btn.textContent = '✓ Connected';
    showToast('Connection request sent! 🤝');
  }

  function filterChip(el) {
    document.querySelectorAll('.filter-chips .chip').forEach(c => c.classList.remove('active'));
    el.classList.add('active');
  }

  let toastTimeout;
  function showToast(msg) {
    const toast = document.getElementById('toast');
    toast.textContent = msg;
    toast.classList.add('show');
    clearTimeout(toastTimeout);
    toastTimeout = setTimeout(() => toast.classList.remove('show'), 2500);
  }
</script>
